030package com.tibco.ep.buildmavenplugin;
031
032import org.apache.maven.artifact.Artifact;
033import org.apache.maven.plugin.MojoExecutionException;
034import org.apache.maven.plugins.assembly.model.Assembly;
035import org.apache.maven.plugins.assembly.model.FileItem;
036import org.apache.maven.plugins.assembly.model.FileSet;
037import org.apache.maven.plugins.annotations.Mojo;
038import org.apache.maven.plugins.annotations.Parameter;
039import org.apache.maven.plugins.annotations.ResolutionScope;
040
041import static org.apache.maven.plugins.annotations.LifecyclePhase.PACKAGE;
042
043import java.io.File;
044import java.util.ArrayList;
045import java.util.HashMap;
046import java.util.List;
047import java.util.Map;
048
049/**
050 * <p>Build a LiveView fragment</p>
051 * 
052 * <p>The packaging rules are as follows :-</p>
053 * <ol>
054 * <li>A mainifest file is created at /META-INF/MANIFEST.MF containing :-
055 *      <ul>
056 *              <li>Archiver-Version: Plexus Archiver</li>
057 *              <li>Built-By: build user</li>
058 *              <li>Build-Jdk: jdk version</li>
059 *              <li>Package-Title: project groupId and name</li>
060 *              <li>Package-Version: project version and buildNumber</li>
061 *              <li>Package-Vendor: project organization name (if set)</li>
062 *      </ul>
063 * <li>The project's files are copied into /</li>
064 * <li>The project's java classes and jar dependencies are copied into /java-resources</li>
065 * <li>The project's resources files are copied into /</li>
066 * </ol>
067 * 
068 * <p>The plexus archiver is used to create the archive via the maven assembly
069 * plugin.</p>
070 * 
071 * <p>Native archives (nar) are included - in this case a mapping is included
072 * from the nar AOL values (http://maven-nar.github.io/aol.html) to internal
073 * values.</p>
074 * 
075 * <p>The generated filename is &lt;artifactId&gt;-&lt;version&gt;-ep-liveview-fragment.zip</p>
076 * 
077 */
078@Mojo(name = "package-liveview-fragment", defaultPhase = PACKAGE, requiresDependencyResolution = ResolutionScope.RUNTIME)
079public class PackageLiveViewFragmentMojo extends BasePackageMojo {
080
081    /**
082     * <p>Eventflow source directories</p>
083     * 
084     * <p>If no eventflowDirectories is specified, a single directory of
085     * ${project.basedir}/src/main/eventflow is used.</p>
086     * 
087     * <p>Example use in pom.xml:</p>
088     * <img src="uml/eventflowDirectories.svg" alt="pom">
089     * 
090     * <p>Example use on commandline:</p>
091     * <img src="uml/eventflowDirectories-commandline.svg" alt="pom">
092     * 
093     * @since 1.0.0
094     */
095    @Parameter( required = false, property = "eventflowDirectories" )
096    File[] eventflowDirectories;
097
098    /**
099     * <p>Liveview source directory</p>
100     * 
101     * <p>Example use in pom.xml:</p>
102     * <img src="uml/liveviewDirectory.svg" alt="pom">
103     * 
104     * <p>Example use on commandline:</p>
105     * <img src="uml/liveviewDirectory-commandline.svg" alt="pom">
106     * 
107     * @since 1.0.0
108     */
109    @Parameter( defaultValue = "${project.basedir}/src/main/liveview", required = true, property = "liveviewDirectory" )
110    File liveviewDirectory;
111
112    /**
113     * <p>Additional resources directory for HOCON configurations</p>
114     * 
115     * <p>This is added to the list of resource directories</p>
116     * 
117     * <p>Example use in pom.xml:</p>
118     * <img src="uml/package-configurationDirectory.svg" alt="pom">
119     * 
120     * @since 1.0.0
121     */
122    @Parameter( defaultValue = "${project.basedir}/src/main/configurations", required = true )
123    File configurationDirectory;
124
125    /**
126     * <p>Additional resources directory for test HOCON configurations</p>
127     * 
128     * <p>This is added to the list of test resource directories</p>
129     * 
130     * <p>Example use in pom.xml:</p>
131     * <img src="uml/package-testConfigurationDirectory.svg" alt="pom">
132     * 
133     * @since 1.0.0
134     */
135    @Parameter( defaultValue = "${project.basedir}/src/test/configurations", required = true )
136    File testConfigurationDirectory;
137
138
139    public void execute() throws MojoExecutionException {
140        getLog().debug( "Creating liveview fragment" );
141
142        if (eventflowDirectories == null || eventflowDirectories.length == 0) {
143            eventflowDirectories = new File[] { new File(project.getBasedir(), "/src/main/eventflow") };
144        }
145
146        prechecks();
147
148        Assembly assembly = createAssembly();
149
150        // add runtime dependencies
151        //
152        // we add them one-by-one instead of using a dependencyset so we can
153        // set the classpath in the manifest file
154        //
155        String classPath = "";
156        String nativeClassPath = "";
157        String fragmentList = "";
158
159        List<FileItem> deps = new ArrayList<FileItem>();
160        for (Artifact dependency : getCompileProjectDependencies()) {
161            String type = dependency.getType();
162            String destName = dependency.getGroupId()+"-"+dependency.getArtifactId()+"-"+dependency.getBaseVersion();
163
164            if (type != null && (type.equals(JAVA_TYPE) || type.equals(EVENTFLOW_TYPE) || type.equals(TCS_TYPE) || type.equals(LIVEVIEW_TYPE))) {
165                destName += "-"+type+".zip";
166            } else {
167                if (dependency.hasClassifier()) {
168                    destName += "-"+dependency.getClassifier();
169                }
170                destName += "."+dependency.getType();
171            }
172
173            FileItem dependencyFile = new FileItem();
174            dependencyFile.setOutputDirectory("");
175            dependencyFile.setSource(getArtifactPath(dependency));
176            dependencyFile.setDestName(destName);
177            deps.add(dependencyFile);
178
179            if (dependency.getType().equals("nar")) {
180                nativeClassPath += destName+" ";
181            } else if (type != null && (type.equals(JAVA_TYPE) || type.equals(EVENTFLOW_TYPE) || type.equals(TCS_TYPE) || type.equals(LIVEVIEW_TYPE))) {
182                fragmentList += destName+" ";
183            } else {
184                classPath += destName+" ";
185            }
186        }
187
188        // add files in our source directories
189        //
190        for (File eventflowDirectory : eventflowDirectories) {
191            FileSet fileSet = new FileSet();
192            fileSet.setDirectory(eventflowDirectory.getAbsolutePath());
193            fileSet.setOutputDirectory("");
194            assembly.addFileSet(fileSet);
195        }
196        FileSet fileSet = new FileSet();
197        fileSet.setDirectory(liveviewDirectory.getAbsolutePath());
198        fileSet.setOutputDirectory("");
199        assembly.addFileSet(fileSet);
200        
201        // build a manifest file and add as first file to the zip
202        //
203        Map<String, String> extras = new HashMap<String, String>();
204        extras.put("TIBCO-EP-Fragment-Format-Version", "2");
205        if (!classPath.isEmpty()) {
206            extras.put("Class-Path", classPath);
207        }
208        if (!nativeClassPath.isEmpty()) {
209            extras.put("TIBCO-EP-Native-Class-Path", nativeClassPath);
210        }
211        if (!fragmentList.isEmpty()) {
212            extras.put("TIBCO-EP-Fragment-List", fragmentList);
213        }            
214        File manifest = packageManifest(assembly, null, extras);
215
216        for (FileItem dependencyFile : deps) {
217            assembly.addFile(dependencyFile);
218        }
219
220        // add output directory
221        //
222        // this should include resources ( already copied here ) and
223        // eventflow files
224        //
225        fileSet = new FileSet();
226        fileSet.setDirectory(project.getBuild().getOutputDirectory());
227        fileSet.setOutputDirectory("");
228        assembly.addFileSet(fileSet);
229
230        // write assembly
231        //
232        writeAssembly(assembly);
233        if (!manifest.delete()) {
234            getLog().debug("Unable to delete "+manifest.getAbsolutePath());
235        }
236
237    }
