030package com.tibco.ep.buildmavenplugin;
031
032import java.io.File;
033import java.util.Properties;
034
035import org.apache.maven.model.PluginExecution;
036import org.apache.maven.plugin.MojoExecutionException;
037import org.apache.maven.plugins.annotations.Mojo;
038import org.apache.maven.plugins.annotations.Parameter;
039
040
041/**
042 * <p>Stop and remove nodes.</p>
043 * 
044 * <p>When executed in clean phase, any errors found will be logged but won't 
045 * cause the build to fail.  When executed in a phase other than clean, a stop 
046 * failure will fail the build.</p>
047 */
048@Mojo(name = "stop-nodes", threadSafe = true)
049public class StopNodesMojo extends BaseExecuteMojo {
050
051    /**
052     * <p>Set this to 'true' to skip stopping test nodes</p>
053     * 
054     * <p>Example use in pom.xml:</p>
055     * <img src="uml/start-nodes-skipStop.svg" alt="pom">
056     * 
057     * <p>Example use on commandline:</p>
058     * <img src="uml/start-nodes-skipStop-commandline.svg" alt="pom">
059     * 
060     * @since 1.0.0
061     */
062    @Parameter(property = "skipStop")
063    boolean skipStop;
064
065    /**
066     * <p>Set this to 'true' to skip running tests, but still compile them.</p>
067     * 
068     * <p>Example use in pom.xml:</p>
069     * <img src="uml/start-nodes-skipTests.svg" alt="pom">
070     * 
071     * <p>Example use on commandline:</p>
072     * <img src="uml/start-nodes-skipTests-commandline.svg" alt="pom">
073     * 
074     * @since 1.0.0
075     */
076    @Parameter(property = "skipTests")
077    boolean skipTests;
078    
079    /**
080     * <p>Set this to 'true' to skip removing test nodes</p>
081     * 
082     * <p>Example use in pom.xml:</p>
083     * <img src="uml/stop-nodes-skipRemove.svg" alt="pom">
084     * 
085     * <p>Example use on commandline:</p>
086     * <img src="uml/stop-nodes-skipRemove-commandline.svg" alt="pom">
087     * 
088     * @since 1.0.0
089     */
090    @Parameter(property = "skipRemove", defaultValue="false")
091    boolean skipRemove;
092
093    /**
094     * <p>Set this to 'true' to only install nodes</p>
095     * 
096     * <p>Example use in pom.xml:</p>
097     * <img src="uml/start-nodes-installOnly.svg" alt="pom">
098     * 
099     * <p>Example use on commandline:</p>
100     * <img src="uml/start-nodes-installOnly-commandline.svg" alt="pom">
101     * 
102     * @since 1.1.0
103     */
104    @Parameter(property = "installOnly")
105    boolean installOnly;
106    
107    /**
108     * Set this to 'false' to ignore errors on node stop
109     * 
110     * <p>Example use in pom.xml:</p>
111     * <img src="uml/stop-nodes-failOnStopError.svg" alt="pom">
112     * 
113     * <p>Example use on commandline:</p>
114     * <img src="uml/stop-nodes-failOnStopError-commandline.svg" alt="pom">
115     * 
116     * @since 1.0.0
117     */
118    @Parameter(property = "failOnStopError", defaultValue="true")
119    boolean failOnStopError;
120    
121    /**
122     * <p>List of node names to start ready for test cases to run.  If not set, a
123     * single node <b>A</b> is used.</p>
124     * 
125     * <p>Example use in pom.xml:</p>
126     * <img src="uml/start-nodes-nodes.svg" alt="pom">
127     * 
128     * @since 1.0.0
129     */
130    @Parameter
131    String[] nodes = new String[] {"A"};
132
133    /**
134     * Installation path - overrides servicename, nodes are ignored
135     *
136     * @since 1.6.0
137     */
138    @Parameter
139    String installPath;
140    
141    public void execute() throws MojoExecutionException {
142        getLog().debug("Stop nodes");
143
144        prechecks();
145
146        Properties modelProperties = project.getModel().getProperties();
147        boolean testCasesFound = true;
148        if (modelProperties.getProperty(TESTCASESFOUND_PROPERTY) != null && modelProperties.getProperty(TESTCASESFOUND_PROPERTY).equals("false")) {
149            testCasesFound = false;
150        }
151        
152        boolean clean = false;
153        if (execution.getLifecyclePhase() != null && (execution.getLifecyclePhase().equals("clean") || execution.getLifecyclePhase().equals("pre-clean"))) {
154            clean = true;
155        }
156        
157        // determine if we have executions steps in the pom - if so we can skip id this
158        // run is a default one
159        //
160        boolean hasExecutions = false;
161        if (mojoExecution.getPlugin() != null && mojoExecution.getPlugin().getExecutions() != null) {
162            for (PluginExecution p : mojoExecution.getPlugin().getExecutions()) {
163                if (!p.getId().startsWith("default-")) {
164                    hasExecutions = true;
165                }
166            }
167        }
168        
169        if (!clean && (
170                (hasExecutions && mojoExecution.getExecutionId().startsWith("default-")) || skipTests || installOnly || !testCasesFound)) {
171            getLog().info("Stop and remove nodes is skipped (skipTests="+skipTests+",hasExecutions="+hasExecutions+",testCasesFound="+testCasesFound+",installOnly="+installOnly+",phase="+execution.getLifecyclePhase()+")");
172            return;
173        }
174
175        boolean failOnError = failOnStopError;
176        if (clean) {
177            failOnError = false;
178        }
179
180        initializeAdministration(failOnError);
181
182        // shut down
183        //
184        if (skipStop) {
185            getLog().info("Stop nodes is skipped");
186        } else {
187            for (int i=0; i<nodes.length; i++) {
188                String nodeName = nodes[i]+"."+clusterName;
189                if (new File(nodeDirectory, nodeName).exists()) {
190                    stopNodes(nodeName, userName, password, failOnError);
191                }
192            }
193        }
194
195        if (skipRemove) {
196            getLog().info("Remove nodes is skipped");
197            return;
198        }
199
200        // remove nodes
201        //
202        if (installPath != null) {
203            removeNode(installPath, failOnError);
204        } else {
205            for (int i=0; i<nodes.length; i++) {
206                String nodeName = nodes[i]+"."+clusterName;
207                if (new File(nodeDirectory, nodeName).exists()) {
208                    try {
209                        removeNodes(nodeName, userName, password, failOnError);
210                    } catch (MojoExecutionException e) {
211                        removeNode(new File(nodeDirectory, nodeName).getAbsolutePath(), failOnError);
212                    }
213                }
214            }
215        }
216
217    }
218}
