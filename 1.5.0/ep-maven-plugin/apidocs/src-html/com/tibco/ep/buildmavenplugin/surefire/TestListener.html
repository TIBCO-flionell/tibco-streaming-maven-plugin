030package com.tibco.ep.buildmavenplugin.surefire;
031
032import java.util.ResourceBundle;
033import org.apache.maven.surefire.Surefire;
034import org.apache.maven.surefire.report.PojoStackTraceWriter;
035import org.apache.maven.surefire.report.ReportEntry;
036import org.apache.maven.surefire.report.ReporterException;
037import org.apache.maven.surefire.report.ReporterManager;
038import org.junit.runner.Description;
039import org.junit.runner.Result;
040import org.junit.runner.notification.Failure;
041import org.junit.runner.notification.RunListener;
042
043/**
044 * JUnit run listener that republishes JUnit events to the report manager.
045 */
046public class TestListener extends RunListener {
047    /**
048     * Constructor.
049     *
050     * @param testSet       the specific test set that this will report on as it is
051     *                      executed
052     * @param reportManager the report manager to log testing events to
053     */
054    TestListener(TestSet testSet, ReporterManager reportManager) {
055        m_testSet = testSet;
056        m_reportMgr = reportManager;
057        m_testFailed = false;
058    }
059
060    /**
061     * Test run started
062     * @param description Test description
063     * @throws ReporterException Reporting error
064     */
065    @Override
066    public void testRunStarted(Description description) throws ReporterException {
067        ReportEntry report = new ReportEntry(
068                m_testSet.getName(),
069                m_testSet.getName(),
070                bundle.getString("testSetStarting"));
071
072        m_reportMgr.testSetStarting(report);
073    }
074
075    /**
076     * Test run finished
077     * @param result Test result
078     */
079    @Override
080    public void testRunFinished(Result result) {
081        ReportEntry report = new ReportEntry(
082                m_testSet.getName(),
083                m_testSet.getName(),
084                bundle.getString("testSetCompletedNormally"));
085
086        this.m_reportMgr.testSetCompleted( report );
087        this.m_reportMgr.reset();
088    }
089
090    /**
091     * Test ignored
092     * @param description Test description
093     */
094    @Override
095    public void testIgnored(Description description) {
096        ReportEntry report = new ReportEntry(
097                m_testSet.getClass().getName(),
098                description.getDisplayName(),
099                bundle.getString("testSkipped") );
100
101        m_reportMgr.testSkipped(report);
102    }
103
104    /**
105     * Test started
106     * @param description Test description
107     * @throws Exception Test initialization faiulre
108     */
109    @Override
110    public void testStarted(Description description)
111        throws Exception {
112        ReportEntry report = new ReportEntry(
113                m_testSet.getClass().getName(),
114                description.getDisplayName(),
115                bundle.getString("testStarting"));
116
117        m_reportMgr.testStarting(report);
118        m_testFailed = false;
119    }
120
121    /**
122     * Test failed
123     * @param failure Failure reason
124     */
125    @Override
126    public void testFailure(Failure failure) {
127        ReportEntry report = new ReportEntry(
128                m_testSet.getClass().getName(),
129                failure.getTestHeader(),
130                bundle.getString( "executeException" ),
131                new PojoStackTraceWriter( 
132                    m_testSet.getClass().getName(),
133                    m_testSet.getName(),
134                    failure.getException()));
135
136        if (failure.getException() instanceof AssertionError) {
137            m_reportMgr.testFailed( report );
138        } else {
139            m_reportMgr.testError( report );
140        }
141
142        m_testFailed = true;
143    }
144
145    /**
146     * Test finished
147     * @param description Test description
148     */
149    @Override
150    public void testFinished(Description description) {
151        if ( m_testFailed == false ) {
152            ReportEntry report = new ReportEntry(
153                    m_testSet.getName(),
154                    description.getDisplayName(),
155                    bundle.getString( "testSuccessful" ));
156
157            this.m_reportMgr.testSucceeded( report );
158        }
159    }
160
161    private static final ResourceBundle bundle
162            = ResourceBundle.getBundle(Surefire.SUREFIRE_BUNDLE_NAME);
163    private final TestSet m_testSet;
164    private final ReporterManager m_reportMgr;
165
166    /**
167     * This flag is set after a failure has occurred so that a
168     * <code>testSucceeded</code> event is not fired. This is necessary because
169     * JUnit4 always fires a <code>testRunFinished</code> event-- even if
170     * there was a failure.
171     */
172    private boolean m_testFailed;
173}
