030package com.tibco.ep.buildmavenplugin;
031
032import org.apache.maven.plugin.MojoExecutionException;
033import org.apache.maven.plugins.annotations.LifecyclePhase;
034import org.apache.maven.plugins.annotations.Mojo;
035import org.apache.maven.plugins.annotations.Parameter;
036
037import java.io.File;
038import java.io.IOException;
039import java.nio.file.FileVisitResult;
040import java.nio.file.Files;
041import java.nio.file.Path;
042import java.nio.file.Paths;
043import java.nio.file.SimpleFileVisitor;
044import java.nio.file.StandardCopyOption;
045import java.nio.file.attribute.BasicFileAttributes;
046
047/**
048 * <p>Compile eventflow</p>
049 * 
050 * <p>Compilation includes copying the source files to the output directory</p>
051 */
052@Mojo(name = "compile-liveview-fragment", defaultPhase = LifecyclePhase.COMPILE, threadSafe = true)
053public class CompileLiveView extends BaseMojo {
054    
055    /**
056     * <p>Liveview source directory</p>
057     * 
058     * <p>Example use in pom.xml:</p>
059     * <img src="uml/liveviewDirectory.svg" alt="pom">
060     * 
061     * @since 1.0.0
062     */
063    @Parameter( defaultValue = "${project.basedir}/src/main/liveview", required = true, property = "liveviewDirectory" )
064    File liveviewDirectory;
065    
066    
067    public void execute() throws MojoExecutionException {
068        
069        getLog().debug( "Compiling liveview fragment "+liveviewDirectory);
070
071        prechecks();
072        
073        if (liveviewDirectory.exists()) {
074            try {
075                Files.walkFileTree(liveviewDirectory.toPath(), new SimpleFileVisitor<Path>() {
076
077                    @Override
078                    public FileVisitResult preVisitDirectory(Path dir, BasicFileAttributes attrs) {
079                        try {
080                            Files.createDirectories(Paths.get(project.getBuild().getOutputDirectory(), liveviewDirectory.toPath().relativize(dir).toString()));
081                        } catch (IOException e) {
082                            // ignore
083                        }
084                        return FileVisitResult.CONTINUE;
085                    }
086
087                    @Override
088                    public FileVisitResult visitFile(Path file, BasicFileAttributes attrs) throws IOException {
089                        if (file.toString().endsWith(".lvconf")) {
090
091                            // copy file to output directory
092                            //                                
093                            Files.copy(file, Paths.get(project.getBuild().getOutputDirectory(), liveviewDirectory.toPath().relativize(file).toString()), StandardCopyOption.REPLACE_EXISTING);
094
095                        }
096                        return FileVisitResult.CONTINUE;
097                    }
098                });
099            } catch (IOException e) {
100                throw new MojoExecutionException("Failed to read resource directory: " + e.getMessage(), e);
101            }
102        }
103    }
104
105}
