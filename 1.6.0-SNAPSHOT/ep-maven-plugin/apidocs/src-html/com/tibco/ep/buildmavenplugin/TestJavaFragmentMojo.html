030package com.tibco.ep.buildmavenplugin;
031
032import org.apache.maven.model.PluginExecution;
033import org.apache.maven.model.Resource;
034import org.apache.maven.plugin.MojoExecutionException;
035import org.apache.maven.plugins.annotations.Mojo;
036import org.apache.maven.plugins.annotations.Parameter;
037
038import static org.apache.maven.plugins.annotations.LifecyclePhase.TEST;
039
040import java.io.File;
041import java.util.Properties;
042
043/**
044 * <p>Test a java fragment using junit and surefire.</p>
045 * 
046 * <p>A java runner for surefire is created and deployed to the test nodes - 
047 * this runner invokes surefire and hence junit to run the test cases.</p>
048 * 
049 * <p>The java runner sets the system property <b>net.sourceforge.cobertura.datafile</b>
050 * to a unique file in the build directory.  So when coverage is enabled, these
051 * per-node reports are merged into a single report for processing.</p>
052 * 
053 * <p>The java runner takes the configured reportsDirectory property and
054 * appends the node name - the node specific test report files are created in
055 * this directory.  When all the test have finished, these node specific test
056 * reports are renamed into the configured reportsDirectory directory.</p>
057 * 
058 * <p>The maven property jenkins.executionId.reportsDirectory is set so that
059 * jenkins can locate the test reports.</p>
060 * 
061 * <p>Java assertions are always enabled.  Options file is always ignored.</p>
062 * 
063 * <p>The following system properties are set :</p>
064 * <ul>
065 * <li><b>com.tibco.ep.dtm.fragment.version</b> - fragment version</li>
066 * <li><b>com.tibco.ep.dtm.fragment.identifier</b> - fragment identifier</li>
067 * </ul>
068 */
069@Mojo(name = "test-java-fragment", defaultPhase = TEST, threadSafe = true)
070public class TestJavaFragmentMojo extends BaseTestMojo {
071    
072    /**
073     * <p>Additional resources directory for HOCON configurations</p>
074     * 
075     * <p>This is added to the list of resource directories</p>
076     * 
077     * <p>Example use in pom.xml:</p>
078     * <img src="uml/package-configurationDirectory.svg" alt="pom">
079     * 
080     * @since 1.0.0
081     */
082    @Parameter( defaultValue = "${project.basedir}/src/main/configurations", required = true )
083    File configurationDirectory;
084
085    /**
086     * <p>Additional resources directory for test HOCON configurations</p>
087     * 
088     * <p>This is added to the list of test resource directories</p>
089     * 
090     * <p>Example use in pom.xml:</p>
091     * <img src="uml/package-testConfigurationDirectory.svg" alt="pom">
092     * 
093     * @since 1.0.0
094     */
095    @Parameter( defaultValue = "${project.basedir}/src/test/configurations", required = true )
096    File testConfigurationDirectory;
097    
098    public void execute() throws MojoExecutionException {
099        getLog().debug("Testing Java fragment");
100
101        Properties modelProperties = project.getModel().getProperties();
102        boolean testCasesFound = true;
103        if (modelProperties.getProperty(TESTCASESFOUND_PROPERTY) != null && modelProperties.getProperty(TESTCASESFOUND_PROPERTY).equals("false")) {
104            testCasesFound = false;
105        }
106
107        // determine if we have executions steps in the pom - if so we can skip id this
108        // run is a default one
109        //
110        boolean hasExecutions = false;
111        if (mojoExecution.getPlugin() != null && mojoExecution.getPlugin().getExecutions() != null) {
112            for (PluginExecution p : mojoExecution.getPlugin().getExecutions()) {
113                if (!p.getId().startsWith("default-")) {
114                    hasExecutions = true;
115                }
116            }
117        }
118        
119        if ((hasExecutions && mojoExecution.getExecutionId().startsWith("default-")) || skipTests || installOnly || !testCasesFound) {
120            getLog().info("Tests are skipped (skipTests="+skipTests+",hasExecutions="+hasExecutions+",installOnly="+installOnly+",testCasesFound="+testCasesFound+")");
121            return;
122        }
123        
124        prechecks();
125        
126        Resource resource = new Resource();
127        resource.setDirectory(configurationDirectory.getAbsolutePath());
128        resource.setTargetPath("configurations");
129        project.getBuild().addResource(resource);
130        resource = new Resource();
131        resource.setDirectory(testConfigurationDirectory.getAbsolutePath());
132        resource.setTargetPath("test-configuration");
133        project.getBuild().addTestResource(resource);  
134        
135        initializeAdministration(true);
136        
137        // run test cases
138        //
139        runJunitTests(true, null, null);
140
141    }
142
143}
