030package com.tibco.ep.buildmavenplugin;
031
032import org.apache.maven.artifact.Artifact;
033import org.apache.maven.plugin.MojoExecutionException;
034import org.apache.maven.plugins.annotations.Mojo;
035import org.apache.maven.plugins.annotations.Parameter;
036
037import org.apache.maven.plugins.assembly.model.Assembly;
038import org.apache.maven.plugins.assembly.model.FileItem;
039import org.apache.maven.plugins.assembly.model.FileSet;
040
041import static org.apache.maven.plugins.annotations.LifecyclePhase.PACKAGE;
042
043import java.io.File;
044import java.util.ArrayList;
045import java.util.HashMap;
046import java.util.List;
047import java.util.Map;
048
049/**
050 * <p>Build a java fragment zip archive.</p>
051 *
052 * <p>The packaging rules are as follows :-</p>
053 * <ol>
054 * <li>A mainifest file is created at /META-INF/MANIFEST.MF containing :-
055 *      <ul>
056 *              <li>Archiver-Version: Plexus Archiver</li>
057 *              <li>Built-By: build user</li>
058 *              <li>Build-Jdk: jdk version</li>
059 *              <li>Package-Title: project groupId and name</li>
060 *              <li>Package-Version: project version and buildNumber</li>
061 *              <li>Package-Vendor: project organization name (if set)</li>
062 *              <li>Main-Class: mainClass parameter (if set)</li>
063 *      </ul>
064 * <li>pom.xml copied to /META-INF/maven/groupId/artifactId/pom.xml</li>
065 * <li>pom.properties created in /META-INF/maven/groupId/artifactId/pom.properties</li>
066 * <li>The project's built java classes files are copied into /classes</li>
067 * <li>The project's resource files are copied into /</li>
068 * <li>Java runtime dependencies are copied into /dependencies/java/</li>
069 * </ol>
070 * 
071 * <p>The plexus archiver is used to create the archive via the maven assembly
072 * plugin.</p>
073 * 
074 * <p>Native archives (nar) are included - in this case a mapping is included
075 * from the nar AOL values (http://maven-nar.github.io/aol.html) to internal
076 * values.</p>
077 * 
078 * <p>The generated filename is &lt;artifactId&gt;-&lt;version&gt;-ep-java-fragment.zip</p>
079 */
080@Mojo(name = "package-java-fragment", defaultPhase = PACKAGE)
081public class PackageJavaFragmentMojo extends BasePackageMojo {
082
083    // maven user parameters
084    //
085
086    /**
087     * <p>Java Main class name.  This will be added to the MANIFEST file.</p>
088     * 
089     * <p>Example use in pom.xml:</p>
090     * <img src="uml/package-mainClass.svg" alt="pom">
091     * 
092     * @since 1.0.0
093     */
094    @Parameter
095    String mainClass;
096
097    public void execute() throws MojoExecutionException {
098        getLog().debug( "Creating fragment zip" );
099
100        prechecks();
101
102        // updated format which is really jar
103        //
104
105        Assembly assembly = createAssembly();
106
107        // add runtime dependencies
108        //
109        // we add them one-by-one instead of using a dependencyset so we can
110        // set the classpath in the manifest file
111        //
112        String classPath = "";
113        String nativeClassPath = "";
114        String fragmentList = "";
115
116        List<FileItem> deps = new ArrayList<FileItem>();
117        for (Artifact dependency : getCompileProjectDependencies()) {
118            String type = dependency.getType();
119            String destName = dependency.getGroupId()+"-"+dependency.getArtifactId()+"-"+dependency.getBaseVersion();
120
121            if (type != null && (type.equals(JAVA_TYPE) || type.equals(EVENTFLOW_TYPE) || type.equals(TCS_TYPE) || type.equals(LIVEVIEW_TYPE))) {
122                destName += "-"+type+".zip";
123            } else {
124                if (dependency.hasClassifier()) {
125                    destName += "-"+dependency.getClassifier();
126                }
127                destName += "."+dependency.getType();
128            }
129
130            FileItem dependencyFile = new FileItem();
131            dependencyFile.setOutputDirectory("");
132            dependencyFile.setSource(getArtifactPath(dependency));
133            dependencyFile.setDestName(destName);
134            deps.add(dependencyFile);
135
136            if (dependency.getType().equals("nar")) {
137                nativeClassPath += destName+" ";
138            } else if (type != null && (type.equals(JAVA_TYPE) || type.equals(EVENTFLOW_TYPE) || type.equals(TCS_TYPE) || type.equals(LIVEVIEW_TYPE))) {
139                fragmentList += destName+" ";
140            } else {
141                classPath += destName+" ";
142            }
143        }
144
145        // build a manifest file and add as first file to the zip
146        //
147        Map<String, String> extras = new HashMap<String, String>();
148        extras.put("TIBCO-EP-Fragment-Format-Version", "2");
149        if (!classPath.isEmpty()) {
150            extras.put("Class-Path", classPath);
151        }
152        if (!nativeClassPath.isEmpty()) {
153            extras.put("TIBCO-EP-Native-Class-Path", nativeClassPath);
154        }
155        if (!fragmentList.isEmpty()) {
156            extras.put("TIBCO-EP-Fragment-List", fragmentList);
157        }
158        File manifest = packageManifest(assembly, mainClass, extras);
159
160        for (FileItem dependencyFile : deps) {
161            assembly.addFile(dependencyFile);
162        }
163
164        // add output directory
165        //
166        // this should include resources ( already copied here ) and
167        // eventflow files
168        //
169        FileSet fileSet = new FileSet();
170        fileSet.setDirectory(project.getBuild().getOutputDirectory());
171        fileSet.setOutputDirectory("");
172        assembly.addFileSet(fileSet);
173
174        // write assembly
175        //
176        writeAssembly(assembly);
177        if (!manifest.delete()) {
178            getLog().debug("Unable to delete "+manifest.getAbsolutePath());
179        }
180    }
181
182}
