030package com.tibco.ep.buildmavenplugin;
031
032import java.io.File;
033
034import org.apache.maven.plugin.testing.MojoRule;
035import org.apache.maven.plugin.testing.resources.TestResources;
036import org.junit.Assert;
037import org.junit.Rule;
038import org.junit.Test;
039
040import org.slf4j.Logger;
041import org.slf4j.LoggerFactory;
042
043import com.tibco.ep.buildmavenplugin.AdministrationMojo;
044import com.tibco.ep.buildmavenplugin.InstallProductMojo;
045import com.tibco.ep.buildmavenplugin.StartNodesMojo;
046import com.tibco.ep.buildmavenplugin.StopNodesMojo;
047
048/**
049 * Node lifecycle - start and stop
050 *
051 */
052public class NodeLifecycleTest extends BetterAbstractMojoTestCase  {
053    private Logger logger = LoggerFactory.getLogger(NodeLifecycleTest.class);
054
055    /**
056     * rule
057     */
058    @Rule
059    public MojoRule rule = new MojoRule();
060
061    /**
062     * resources
063     */
064    @Rule
065    public TestResources resources = new TestResources();
066
067    /**
068     * Install product - do this first
069     * 
070     * @throws Exception on error
071     */
072    @Test
073    public void testNodeLifecycle() throws Exception  {    
074        logger.info("Install Product");
075        SimulatedLog simulatedLog = new SimulatedLog(false);
076
077        File pom = new File( "target/projects", "pom.xml" );
078        Assert.assertNotNull( pom );
079        Assert.assertTrue( pom.exists());
080
081        InstallProductMojo installProduct = (InstallProductMojo) lookupConfiguredMojo(pom, "install-product");
082        Assert.assertNotNull( installProduct );
083        installProduct.setLog(simulatedLog);     
084        installProduct.execute();
085        assertEquals(simulatedLog.getErrorLog(), 0, simulatedLog.getErrorLog().length());
086        assertEquals(simulatedLog.getWarnLog(), 0, simulatedLog.getWarnLog().length());
087
088        logger.info("Test lifecycle");
089        simulatedLog = new SimulatedLog(false);
090
091        pom = new File( "target/projects/lifecycle", "pom.xml" );
092        Assert.assertNotNull(pom);
093        Assert.assertTrue(pom.exists());
094
095        File status = new File( "target/projects/lifecycle", "status.xml" );
096        Assert.assertNotNull(status);
097        Assert.assertTrue(status.exists());
098
099        File status2 = new File( "target/projects/lifecycle", "status2.xml" );
100        Assert.assertNotNull(status);
101        Assert.assertTrue(status.exists());
102        
103        File stop = new File( "target/projects/lifecycle", "stop.xml" );
104        Assert.assertNotNull(status);
105        Assert.assertTrue(status.exists());
106        
107        try {
108            
109            logger.info("   Start nodes");
110            StartNodesMojo startNodes = (StartNodesMojo) lookupConfiguredMojo(pom, "start-nodes");
111            Assert.assertNotNull(startNodes);
112            startNodes.environment = new String[] { "BUILD_ID="+System.getenv("BUILD_ID") };
113            simulatedLog.reset();
114            startNodes.setLog(simulatedLog);
115            startNodes.execute();
116            assertEquals(simulatedLog.getErrorLog(), 0, simulatedLog.getErrorLog().length());
117            assertEquals(simulatedLog.getWarnLog(), 0, simulatedLog.getWarnLog().length());
118            assertTrue(simulatedLog.getInfoLog(), simulatedLog.getInfoLog().contains("Node started"));
119
120            logger.info("   Display nodes");
121            AdministrationMojo administrationMojo = (AdministrationMojo) lookupConfiguredMojo(status, "administer-nodes");
122            Assert.assertNotNull(administrationMojo);
123            administrationMojo.environment = new String[] { "BUILD_ID="+System.getenv("BUILD_ID") };
124            simulatedLog.reset();
125            administrationMojo.setLog(simulatedLog);
126            administrationMojo.execute();
127            assertEquals(simulatedLog.getErrorLog(), 0, simulatedLog.getErrorLog().length());
128            assertEquals(simulatedLog.getWarnLog(), 0, simulatedLog.getWarnLog().length());
129            assertTrue(simulatedLog.getInfoLog(), simulatedLog.getInfoLog().contains("Started"));
130
131            logger.info("   Display nodes2");
132            administrationMojo = (AdministrationMojo) lookupConfiguredMojo(status2, "administer-nodes");
133            Assert.assertNotNull(administrationMojo);
134            administrationMojo.environment = new String[] { "BUILD_ID="+System.getenv("BUILD_ID") };
135            simulatedLog.reset();
136            administrationMojo.setLog(simulatedLog);
137            administrationMojo.execute();
138            assertEquals(simulatedLog.getErrorLog(), 0, simulatedLog.getErrorLog().length());
139            assertEquals(simulatedLog.getWarnLog(), 0, simulatedLog.getWarnLog().length());
140            assertTrue(simulatedLog.getInfoLog(), simulatedLog.getInfoLog().contains("Started"));
141            
142        } finally {
143            
144            logger.info("   Stop nodes");
145            StopNodesMojo stopNodes = (StopNodesMojo) lookupConfiguredMojo(stop, "stop-nodes");
146            Assert.assertNotNull(stopNodes);
147            stopNodes.environment = new String[] { "BUILD_ID="+System.getenv("BUILD_ID") };
148            simulatedLog.reset();
149            stopNodes.setLog(simulatedLog);
150            stopNodes.execute();
151            assertEquals(simulatedLog.getErrorLog(), 0, simulatedLog.getErrorLog().length());
152            assertEquals(simulatedLog.getWarnLog(), 0, simulatedLog.getWarnLog().length());
153            assertTrue(simulatedLog.getInfoLog(), simulatedLog.getInfoLog().contains("Node removed"));
154            
155        }
156    }
157}
